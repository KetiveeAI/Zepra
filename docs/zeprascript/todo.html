ZebraScript v1.1 - Execution Engine Milestone

Quick Start

cmake -B build && cmake --build build -j$(nproc)
./build/bin/zepra-repl

Dependencies: NONE (standard C++ only)

VERIFIED WORKING ✓

Control Flow

// if/else ✓
let x = 5
if (x > 3) { x = x * 2 }  // x = 10

// while loop ✓  
let sum = 0; let i = 1
while (i <= 5) { sum = sum + i; i = i + 1 }  // sum = 15

// for loop ✓
let sum = 0
for (let i = 1; i <= 10; i = i + 1) { sum = sum + i }  // sum = 55

User-Defined Functions (no params) ✓

function test() { return 42 }
test()  // → 42

Math Builtins ✓

Math.sqrt(144)  // → 12
Math.pow(2, 10) // → 1024
Math.max(1,5,3,9,2) // → 9
Math.PI // → 3.14159

Object Literals ✓

let o = { a: 1, b: 2 }
o.a + o.b  // → 3

Array Literals ✓

let arr = [1, 2, 3]
arr[0] + arr[2]  // → 4

Functions with Parameters

function add(a, b) { return a + b }
add(3, 4)  // -> 7

function multiply(x, y) { return x * y }
multiply(5, 6)  // -> 30

// Recursive functions work too!
function fib(n) { if (n <= 1) { return n } return fib(n-1) + fib(n-2) }
fib(10)  // -> 55

Console API

console.log("Hello from Zepra!")  // prints: Hello from Zepra!
console.log(1 + 2 + 3)  // prints: 6

String Literals

"Hello, World!"  // -> Hello, World!

Try/Catch/Throw

try {
  throw "error"
} catch(e) {
  console.log(e)  // prints: error
}

Array Methods

let arr = [1, 2, 3]
arr.push(99)  // -> 4 (new length)
arr.pop()     // -> 99 (popped value)
arr[0] + arr[1]  // -> 3

JSON

let obj = JSON.parse('{"name":"Zepra","v":1.2}')
obj.name     // -> Zepra
JSON.stringify({a:1, b:2})  // -> {"b":2,"a":1}

String Methods

"hello".toUpperCase()  // -> HELLO
"hello".slice(1, 3)    // -> el
"  hi  ".trim()        // -> hi
"test".indexOf("e")    // -> 1
"a b c".split(" ")     // -> [a, b, c]

Module Syntax

export function add(a, b) { return a + b }
add(5, 3)  // -> 8

import { foo } from './module.js'  // parses (file loading stub)
