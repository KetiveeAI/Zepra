Zepra::initialize()

Initialize the ZepraScript engine. Must be called before any other API calls.

main.cpp 
#include <zeprascript/zepra_api.hpp>

int main() {
    Zepra::initialize();
    
    // Use ZepraScript...
    
    Zepra::shutdown();
    return 0;
}
Class

Zepra::Isolate

An isolated JavaScript execution environment with its own heap and garbage collector.

Creating an Isolate 
Zepra::IsolateOptions opts;
opts.maxHeapSize = 256 * 1024 * 1024;  // 256MB
opts.enableJit = true;

auto isolate = Zepra::Isolate::create(opts);
auto context = isolate->createContext();

auto result = context->evaluate("1 + 2");
if (result.isSuccess()) {
    std::cout << result.value().toNumber();  // 3
}

Security

ResourceMonitor

Track and limit resource usage for untrusted code.

#include <zeprascript/runtime/Sandbox.h>

ExecutionLimits limits = ExecutionLimits::browser();
limits.maxHeapBytes = 50 * 1024 * 1024;
limits.maxExecutionTime = std::chrono::seconds(30);

ResourceMonitor monitor(limits);
vm->setSandbox(&monitor);
