Quick Start

Get up and running in minutes.

main.cpp 
#include <zeprascript/zepra_api.hpp>
#include <iostream>

int main() {
    Zepra::initialize();
    
    auto isolate = Zepra::Isolate::create();
    auto context = isolate->createContext();
    
    auto result = context->evaluate("2 + 2");
    
    if (result.isSuccess()) {
        std::cout << "Result: " << result.value().toNumber() << std::endl;
    }
    
    Zepra::shutdown();
    return 0;
}

Sandboxing

Secure execution of untrusted code.

#include <zeprascript/runtime/Sandbox.h>

ExecutionLimits limits;
limits.maxHeapBytes = 50 * 1024 * 1024;      // 50MB
limits.maxStackBytes = 1024 * 1024;           // 1MB
limits.maxExecutionTime = std::chrono::seconds(10);

ResourceMonitor monitor(limits);
vm->setSandbox(&monitor);

Native Functions

Expose C++ functionality to JavaScript.

Value nativePrint(Context* ctx, const std::vector<Value>& args) {
    for (const auto& arg : args) {
        std::cout << arg.toString() << " ";
    }
    return Value::undefined();
}

void setup(Context* context) {
    auto global = context->globalObject();
    global->set("print", Value::createNativeFunction(nativePrint));
}

// JavaScript: print("Hello", "World");
5

DOM Bindings

Connect JavaScript to your UI layer.

#include <zeprascript/browser/DOMBindings.h>

void setupDOM(Context* context, Document* doc) {
    auto global = context->globalObject();
    global->set("document", DOMBindings::wrapDocument(doc));
    global->set("window", DOMBindings::wrapWindow(context));
}

WebAssembly

Load and execute WASM modules.

#include <zeprascript/wasm/WasmModule.h>

auto module = WasmModule::compile(wasmBytes).value();

WasmImports imports;
imports.addFunction("env", "log", [](int32_t val) {
    std::cout << "WASM: " << val << std::endl;
});

auto instance = module->instantiate(imports);
Value result = instance->call("main", {});
